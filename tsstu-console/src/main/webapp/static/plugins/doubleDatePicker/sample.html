DatePicker Demo

一、区间时间段的选择：

开始时间 : 2015-10-15
结束时间 : 2015-11-11


示例代码：(更多参数，参见用户手册)

<script type="text/javascript">
// 使用时需要引入moment.js 时间格式转换 详见官网 http://momentjs.cn/
var beginDate = moment().subtract(6, 'days').format('YYYY-MM-DD');
var nowDate = moment().format('YYYY-MM-DD');
var dateRange = new pickerDateRange('date', {
// 当天时间是否可选
isTodayValid : true,
// 可选时间是否截止到当天时间 默认不可选择当天之后的时间
stopToday : true,
// 不传递，默认为当天时间
startDate : '2015-10-15',
// 不传递，默认为当天时间
endDate : '2015-11-11',
// 起止时间连接符
defaultText : ' 至 ',
// 设置为true，则没有input框和确定、取消按钮
autoSubmit:true
// 日期输入框是否展示
noCalendar: false,
// 日期只能选择当天之后的
fromToday : false,
// 跟在日期选择框后的按钮 e.g向下箭头 触发事件用此ID
inputTrigger : 'input_trigger_demo1', //这个结构完全由业务系统自己定义
success : function(obj) {
//设置回调句柄 比如说ajax提交，赋值等操作
}
});
</script>
